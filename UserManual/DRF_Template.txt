function downloadCsv(csvString) {
  var type = document.getElementById('answerType').value;
  var countermeasure = document.getElementById('tAnswer').value.split(',').join('');
  var blob = new Blob([type + ',' + countermeasure]);
  if (type == '')
{
  alert('Please select a valid Type');
} else if (countermeasure == '') 
{
  alert('Answer & Countermeasure must not be blank');
} else {
  if (window.navigator.msSaveOrOpenBlob){
    window.navigator.msSaveBlob(blob, "filename.csv");
  }
  else {
    var a = window.document.createElement("a");

    a.href = window.URL.createObjectURL(blob, {
      type: "text/plain"
    });
    a.download = "filename.csv";
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
  }
}
}
